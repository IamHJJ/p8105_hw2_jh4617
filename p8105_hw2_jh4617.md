  read_csv("fivethirtyeight_datasets/snp.csv") |>
  janitor::clean_names() |>
  mutate(date = as.Date(date, format = "%m/%d/%y")) |>
  separate(date, into = c("year", "month", "day")) |>
  mutate(year = as.numeric(year), month = as.numeric(month)) |>
